% Warnings and Cautions
\chapter*{WARNINGS, CAUTIONS, AND NOTES}
\addcontentsline{toc}{chapter}{Warnings, Cautions, and Notes}

\WARNING{%
\textbf{GENERAL WEAPON SAFETY}\\[5pt]
This system controls a lethal weapon. Failure to follow safety procedures can result in death or serious injury to personnel. Always maintain positive identification of targets before engaging. Verify no friendly forces are in the line of fire.
}

\vspace{0.5cm}

\WARNING{%
\textbf{ELECTRICAL HAZARD}\\[5pt]
The RCWS operates on 24-28 VDC at up to 40 amperes. Disconnect all power sources before performing any maintenance. Only qualified personnel may service electrical components.
}

\vspace{0.5cm}

\WARNING{%
\textbf{LASER HAZARD}\\[5pt]
The Laser Range Finder emits Class 1M laser radiation at 1550nm. Do not view directly with optical instruments. Ensure range safety protocols are followed during operations.
}

\vspace{0.5cm}

\CAUTION{%
\textbf{EQUIPMENT DAMAGE}\\[5pt]
Operating the gimbal beyond mechanical limits (-20\degree~to +60\degree~elevation) may cause permanent damage to servos and encoders. Always observe gimbal position indicators. Do not force manual movement when power is applied.
}

\vspace{0.5cm}

\CAUTION{%
\textbf{ENVIRONMENTAL PROTECTION}\\[5pt]
Thermal and day cameras are sensitive to extreme temperatures. Operating outside -20\degree C to +60\degree C ambient range may degrade performance or cause damage. Protect system from moisture, dust, and salt spray when not in use.
}

\vspace{0.5cm}

\NOTE{%
Throughout this manual, the following conventions apply:
\begin{itemize}[leftmargin=2em,itemsep=2pt]
\item \textbf{WARNING} indicates hazards that could result in death or serious injury
\item \textbf{CAUTION} indicates hazards that could result in equipment damage
\item \textbf{NOTE} provides additional information or clarification
\item \button{Button X} indicates joystick or control panel buttons
\item \osd{TEXT} indicates on-screen display elements
\end{itemize}
}

\section*{Pre-Operation Checklist}

\begin{checklist}
\item Main Power ON, Station Enable ON
\item All devices connected (System Status check)
\item Cameras operational (day and thermal)
\item Gimbal smooth motion test (azimuth/elevation)
\item Laser Range Finder test
\item Tracking acquisition test
\item Weapon Arm/Safe switches functional
\item Zero active (\osd{Z} indicator on OSD)
\end{checklist}
