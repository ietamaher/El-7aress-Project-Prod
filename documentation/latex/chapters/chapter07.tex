\# LESSON 7: SYSTEM STATUS \& MONITORING

\textbf{Lesson Duration:} 2 hours (Classroom)

\textbf{Learning Objectives:}
\begin{itemize}
\item Access System Status display
\item Interpret device status indicators
\item Identify fault conditions
\item Determine when to escalate to maintenance
\end{itemize}



\section{7.1 ACCESSING SYSTEM STATUS}

\textbf{Procedure}:
\begin{enumerate}
\item Press \textbf{MENU $\checkmark$}
\item Navigate to \textbf{"System Status"}
\item Press \textbf{VAL}
\end{enumerate}

\textbf{Display Sections}:
\begin{itemize}
\item Azimuth/Elevation Servos
\item IMU (Inertial Measurement Unit)
\item Laser Range Finder (LRF)
\item Day/Night Cameras
\item Control Panels
\item Servo Actuator
\item Alarms/Warnings
\end{itemize}

\section{7.4 WHEN TO ESCALATE TO MAINTENANCE}

\textbf{Operator-Level Issues} (fix yourself):
\begin{itemize}
\item $\checkmark$ Reset E-Stop (after clearing danger)
\item $\checkmark$ Power cycle system (soft reboot)
\item $\checkmark$ Switch cameras (if available)
\item $\checkmark$ Adjust zoom/focus
\item $\checkmark$ Re-zero weapon (if needed)
\end{itemize}

\textbf{Maintenance-Level Issues} (notify immediately):
\begin{itemize}
\item $\times$ Device disconnected (and doesn't reconnect after reboot)
\item $\times$ Persistent fault flags
\item $\times$ Servo malfunction (erratic motion, no motion)
\item $\times$ Physical damage observed
\item $\times$ Thermal runaway (temperature >80\degree{}C)
\item $\times$ Unusual noises (grinding, clicking, squealing)
\end{itemize}
