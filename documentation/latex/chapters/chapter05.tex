\section{5.3 ACQUISITION PHASE}

\subsection{ENTERING ACQUISITION MODE}

\textbf{Prerequisites}:
\begin{itemize}
\item Motion mode: Manual
\item Target on-screen and identified
\item Target approximately centered in reticle
\end{itemize}

\textbf{Procedure}:

\textbf{1. Press \button{Button 4} (Track Select)}

\textbf{Result}:
\begin{itemize}
\item System enters \textbf{ACQUISITION} phase
\item Yellow acquisition gate appears (centered on reticle)
\item OSD displays: \textbf{MODE: ACQUISITION}
\item Joystick gimbal control remains active
\item D-Pad now controls gate size
\end{itemize}



\subsection{SIZING THE ACQUISITION GATE}

\textbf{D-Pad Controls}:

\textbf{Best Practices}:
\begin{itemize}
\item \textbf{Vehicles}: Frame entire hull, exclude ground
\item \textbf{Personnel}: Include torso/legs, minimize background
\item \textbf{Moving targets}: Size slightly larger (anticipate motion)

\subsection{REQUESTING LOCK-ON}

\textbf{When Ready}:
\begin{itemize}
\item Target fully visible and framed in gate
\item Target has good contrast against background
\item Target not moving erratically
\end{itemize}

\textbf{Action}: \textbf{Press \button{Button 4}} (second press)

\textbf{Result}:
\begin{itemize}
\item System $\rightarrow$ \textbf{LOCK PENDING} phase
\item Gate color: Yellow $\rightarrow$ Cyan (or flashing)
\item OSD: \textbf{MODE: LOCK PENDING}
\item Tracker initialization begins (0.5-2 seconds)
\end{itemize}

\CAUTION{Do NOT make rapid gimbal movements during Lock Pending. Hold gimbal steady.}



\section{5.4 LOCK PENDING $\rightarrow$ ACTIVE LOCK}

\subsection{LOCK PENDING PHASE}

\textbf{Purpose}: Tracking system initializes on target

\textbf{Duration}: 0.5 to 2 seconds (typically \textasciitilde{}1 second)

\textbf{Operator Action}: \textbf{WAIT} - maintain steady gimbal

\textbf{What Happens}:
\begin{enumerate}
\item System captures reference image of target
\item Initializes tracker algorithm
\item Calculates target features
\item Begins tracking target in video stream
\end{enumerate}

\textbf{Transition Outcomes}:

\textbf{$\checkmark$ SUCCESS}:
\begin{itemize}
\item System $\rightarrow$ \textbf{ACTIVE LOCK} phase
\item Gate color: Green
\item Gate style: Dashed outline
\item OSD: \textbf{MODE: TRACKING} or \textbf{ACTIVE LOCK}
\item Gimbal control switches to automatic
\end{itemize}

\textbf{$\times$ FAILURE} (rare):
\begin{itemize}
\item System $\rightarrow$ \textbf{ACQUISITION} phase (retry)
\item Possible causes: Poor contrast, target too small, target motion
\item Operator: Adjust gate size or abort and restart
\end{itemize}



\subsection{ACTIVE LOCK PHASE}

\textbf{System Behavior}:
\begin{itemize}
\item Tracker follows target at 30 Hz (30 times/second)
\item Gimbal automatically moves to keep target centered
\item Reticle stays on target center
\item \textbf{Joystick axis inputs IGNORED} (tracking in control)
\end{itemize}

\textbf{Visual Indicators}:
\begin{itemize}
\item \textbf{Tracking Gate}: Green dashed outline around target
\item \textbf{OSD}: MODE: TRACKING or ACTIVE LOCK
\item \textbf{Control Panel}: TRACKING light ON (green)
\item \textbf{Track Confidence}: >70\% (good), 50-70\% (marginal), <50\% (poor)
\end{itemize}



\subsection{OPERATOR ROLE DURING ACTIVE LOCK}

\textbf{Primary Role}: \textbf{MONITOR} - system is tracking automatically

\textbf{Monitor For}:

\begin{enumerate}
\item \textbf{Track Quality}:
\end{enumerate}
\begin{itemize}
\item $\checkmark$ Green gate = Good track
\item  Yellow gate = Marginal (may lose soon)
\item Track confidence >70\% (good), <50\% (prepare for coast)
\end{itemize}

\begin{enumerate}
\item \textbf{Target Status}:
\end{enumerate}
\begin{itemize}
\item Target correctly identified (didn't jump to wrong object)
\item Target still valid (meets engagement criteria)
\item Target not obscured or about to be obscured
\end{itemize}

\begin{enumerate}
\item \textbf{Gimbal Position}:
\end{enumerate}
\begin{itemize}
\item Staying within operational limits
\item Not approaching no-traverse zones
\item Elevation within -20\degree{} to +60\degree{}
\end{itemize}

\textbf{Controls Still Active}:
\begin{itemize}
\item $\checkmark$ \button{Button 0} (Master Arm)
\item $\checkmark$ \button{Button 2} (LAC toggle)
\item $\checkmark$ \button{Button 3} (Dead Man Switch)
\item $\checkmark$ \button{Button 4} (double-click abort)
\item $\checkmark$ \button{Button 5} (Fire)
\item $\checkmark$ \button{Button 6}/8 (Zoom) - use cautiously, may affect track
\item $\times$ \button{Button 11}/13 (Mode cycle) - BLOCKED during tracking
\item $\times$ Joystick axes (X/Y) - IGNORED during tracking
\end{itemize}



\section{5.5 COAST MODE}

\subsection{WHEN COAST ACTIVATES}

\textbf{Triggers}:
\begin{itemize}
\item Target temporarily obscured (passes behind object)
\item Target leaves field of view briefly
\item Tracker loses visual lock
\item Dust, smoke, or other obscurants
\end{itemize}

\textbf{System Behavior}:
\begin{itemize}
\item Tracker \textbf{predicts} target position based on last known velocity
\item Gimbal continues to predicted position
\item System attempts to re-acquire target
\item Tracking gate: Green $\rightarrow$ Yellow (dashed)
\end{itemize}

\textbf{Display Changes}:
\begin{itemize}
\item \textbf{OSD}: MODE: COAST or TRACKING (COAST)
\item \textbf{Gate}: Yellow/amber dashed outline
\item \textbf{Warning}: "COASTING - TARGET LOST" may display
\end{itemize}



\subsection{COAST OUTCOMES}

\textbf{Typical Duration}: 1-5 seconds

\textbf{Outcome 1: Target Re-Acquired} ($\checkmark$ Success):
\begin{itemize}
\item Target reappears in field of view
\item Tracker re-locks on target
\item System $\rightarrow$ \textbf{ACTIVE LOCK} phase
\item Gate: Yellow $\rightarrow$ Green
\item Tracking continues normally
\end{itemize}

\textbf{Outcome 2: Coast Timeout} ($\times$ Failure):
\begin{itemize}
\item Target not re-acquired within timeout (\textasciitilde{}5 seconds)
\item System gives up
\item System $\rightarrow$ \textbf{OFF} phase
\item Tracking stops
\item Operator must restart tracking if desired
\end{itemize}

\textbf{Operator Action During Coast}:
\begin{itemize}
\item $\checkmark$ Wait patiently (system attempting re-acquisition)
\item $\checkmark$ Be ready for track to resume
\item $\times$ Do NOT abort prematurely (give system time)
\item $\times$ Do NOT make manual gimbal movements (joystick ignored)
\end{itemize}

\textbf{Abort Coast If}:
\begin{itemize}
\item Target definitely not coming back (destroyed, permanently obscured)
\item Tracker coasting in wrong direction (confused)
\item Mission changed
\end{itemize}



\section{5.6 TRACKING ABORT (EMERGENCY)}

\subsection{WHEN TO ABORT TRACKING}

Abort tracking \textbf{IMMEDIATELY} if:
\begin{itemize}
\item $\times$ Tracking wrong target (friendly, civilian, incorrect target)
\item $\times$ Target no longer valid (fails ROE)
\item $\times$ Safety concern (entering no-fire zone, gimbal obstruction)
\item $\times$ Mission change (new priority, orders to cease)
\item $\times$ Tracking erratic (unexpected gimbal behavior)
\end{itemize}



\subsection{ABORT PROCEDURE}

\textbf{Action}: \textbf{DOUBLE-CLICK BUTTON 4} (<500ms between presses)

\textbf{Effect} (IMMEDIATE):
\begin{enumerate}
\item Tracking \textbf{STOPS}
\item System $\rightarrow$ \textbf{OFF} phase
\item Gimbal holds current position
\item Tracking gate disappears
\item Weapon fire \textbf{INHIBITED} (even if Master Arm engaged)
\item OSD: MODE: MANUAL
\end{enumerate}

\textbf{Timing}:
\begin{itemize}
\item Press 1 $\rightarrow$ Press 2 within \textbf{500 milliseconds} (half-second)
\item \textbf{Too slow} (>500ms): System interprets as two single presses (may restart tracking)
\end{itemize}

\textbf{ CRITICAL}: Practice double-click timing during training until muscle memory established.



\subsection{AFTER ABORT}

\textbf{System State}:
\begin{itemize}
\item Manual mode active
\item Joystick gimbal control restored
\item No tracking active
\end{itemize}

\textbf{Next Actions} (mission-dependent):
\begin{itemize}
\item Re-acquire correct target and restart tracking
\item Return to surveillance mode
\item Engage different target
\item Follow commander's orders
\end{itemize}



\section{5.7 WEAPONS ENGAGEMENT SEQUENCE}

\subsection{PRE-ENGAGEMENT CHECKLIST}

Before engaging, verify \textbf{ALL} items:

\begin{itemize}
\item $\square$ Target positively identified (\textbf{PID})
\item $\square$ Target valid (meets ROE)
\item $\square$ Fire authorization received (if required)
\item $\square$ NOT in no-fire zone (check OSD for zone warnings)
\item $\square$ Friendly forces clear
\item $\square$ Weapon loaded and ready
\item $\square$ Zeroing active (if applicable)
\item $\square$ Environmental parameters set (if applicable)
\item $\square$ Track established (if using tracking)
\end{itemize}

\WARNING{IF ANY ITEM CANNOT BE CHECKED, DO NOT FIRE.**}



\subsection{ENGAGEMENT PROCEDURE (STEP-BY-STEP)}

\subsubsection{STEP 1: Acquire and Track Target}

\begin{enumerate}
\item Detect and identify target (PID mandatory)
\item \textbf{\button{Button 4}} $\rightarrow$ Enter acquisition mode
\item \textbf{D-Pad} $\rightarrow$ Size gate to frame target (10-30\% margin)
\item \textbf{\button{Button 4}} $\rightarrow$ Request lock-on
\item Wait for \textbf{ACTIVE LOCK} (green gate, OSD: TRACKING)
\item Monitor track quality (confidence >70\%, green gate)
\end{enumerate}



\subsubsection{STEP 2: Range Target}

\begin{enumerate}
\item Fire Laser Range Finder (LRF trigger)
\item Wait for range reading (OSD: \textbf{RNG: xxxx m})
\item Verify range reasonable
\item Range used for ballistic calculations (CCIP)
\end{enumerate}

\textbf{Note}: LRF may fire automatically during tracking (configuration-dependent). Verify range displayed.



\subsubsection{STEP 3: Enable Lead Angle Compensation (If Moving Target)}

\textbf{If target is moving} (lateral motion, speed >5 m/s, range >100m):

\begin{enumerate}
\item Hold \textbf{Dead Man Switch} (\button{Button 3})
\item Press \textbf{LAC Toggle} (\button{Button 2})
\item Verify LAC status:
\end{enumerate}
\begin{itemize}
\item $\checkmark$ \textbf{\osd{LEAD ANGLE ON}} (green) = Ready
\item  \textbf{\osd{LEAD ANGLE LAG}} (yellow) = Wait for tracking data
\item $\times$ \textbf{\osd{ZOOM OUT}} (red) = FOV too narrow, zoom out
\end{itemize}
\begin{enumerate}
\item Observe CCIP reticle offset ahead of target (lead point)
\end{enumerate}

\textbf{If target stationary}: LAC not necessary (CCIP at target center)

\textbf{Detailed LAC procedures in Lesson 6.}



\subsubsection{STEP 4: Final Safety Checks}

\begin{enumerate}
\item Verify target still valid
\item Verify tracking active (green gate, good confidence)
\item Check OSD for warnings:
\end{enumerate}
\begin{itemize}
\item $\times$ \textbf{\osd{ZONE VIOLATION}} = DO NOT FIRE
\item $\times$ \textbf{"NO-FIRE ZONE"} = DO NOT FIRE
\item $\checkmark$ No warnings = Clear to fire
\end{itemize}
\begin{enumerate}
\item Verify friendly forces clear
\item Verify backstop (if required)
\end{enumerate}



\subsubsection{STEP 5: Engage Master Arm}

\begin{enumerate}
\item Pull trigger to \textbf{Stage 1} (half-pull) $\rightarrow$ \textbf{Master Arm} (\button{Button 0})
\end{enumerate}
\begin{itemize}
\item OR toggle Master Arm switch on Control Panel
\end{itemize}
\begin{enumerate}
\item Verify \textbf{\osd{ARMED}} indicator light ON (red)
\item OSD may display: \textbf{\osd{WEAPON ARMED}}
\end{enumerate}

\textbf{ WEAPON IS NOW HOT}



\subsubsection{STEP 6: Fire Weapon}

\begin{enumerate}
\item Final aim verification (CCIP on target or lead point)
\item Pull trigger to \textbf{Stage 2} (full-pull) $\rightarrow$ \textbf{Fire} (\button{Button 5})
\item Weapon fires
\item Hold trigger for desired burst:
\end{enumerate}
\begin{itemize}
\item \textbf{Single Shot}: Quick press/release (1 round)
\item \textbf{Burst}: Hold 2-3 seconds (controlled burst)
\item \textbf{Sustained}: Hold longer (use cautiously)
\end{itemize}

\textbf{During Firing}:
\begin{itemize}
\item Tracking keeps reticle on target
\item Gimbal compensates for recoil
\item Observe rounds impacting
\item Adjust fire as needed
\end{itemize}

\textbf{System State}:
\begin{itemize}
\item Tracking phase $\rightarrow$ \textbf{FIRING}
\item OSD: MODE: TRACKING (FIRING)
\item Gate: Green solid outline
\item Enhanced stabilization active
\end{itemize}



\subsubsection{STEP 7: Cease Fire}

\begin{enumerate}
\item Release \textbf{Fire button} (\button{Button 5} / trigger stage 2)
\item Weapon stops firing
\item Release \textbf{Master Arm} (\button{Button 0} / trigger stage 1)
\item Verify \textbf{\osd{ARMED}} indicator OFF
\item Tracking continues (unless aborted)
\end{enumerate}



\subsubsection{STEP 8: Assess Target (BDA)}

\textbf{Battle Damage Assessment}:
\begin{itemize}
\item $\checkmark$ Target destroyed? $\rightarrow$ Stop tracking, report success
\item  Target damaged? $\rightarrow$ Re-engage (repeat Steps 5-7)
\item $\times$ Target missed? $\rightarrow$ Check zeroing/environmental/LAC, re-engage
\item ðŸ”„ Target suppressed? $\rightarrow$ Maintain track, ready to re-engage
\end{itemize}



\subsubsection{STEP 9: Post-Engagement Actions}

\begin{enumerate}
\item If target neutralized: \textbf{Stop tracking} (double-click \button{Button 4})
\item Report engagement results to command
\item Update ammunition count
\item Scan for additional targets
\item Resume surveillance or follow orders
\end{enumerate}



\section{5.8 ENGAGEMENT BEST PRACTICES}

\subsection{TARGET SELECTION FOR TRACKING}

\textbf{Good Targets}:
\begin{itemize}
\item $\checkmark$ High contrast against background
\item $\checkmark$ Clearly defined edges
\item $\checkmark$ Sufficient size (>30 pixels)
\item $\checkmark$ Relatively stable motion
\end{itemize}

\textbf{Difficult Targets}:
\begin{itemize}
\item  Low contrast (camouflaged)
\item  Very small (distant)
\item  Erratic motion (evasive)
\item  Partially obscured
\end{itemize}

\textbf{If Tracking Fails}:
\begin{itemize}
\item Try manual engagement (no tracking)
\item Improve contrast (switch camera or thermal LUT)
\item Wait for better tracking opportunity
\end{itemize}



\subsection{LEAD ANGLE COMPENSATION TIPS}

\textbf{When to Use LAC}:
\begin{itemize}
\item $\checkmark$ Target moving laterally (crossing FOV)
\item $\checkmark$ Target speed >5 m/s (\textasciitilde{}10 mph)
\item $\checkmark$ Range >100 meters
\end{itemize}

\textbf{When NOT Needed}:
\begin{itemize}
\item $\times$ Stationary targets
\item $\times$ Targets moving radially (toward/away from you)
\item $\times$ Very close range (<50m)
\end{itemize}

\textbf{LAC Limitations}:
\begin{itemize}
\item Requires tracking active
\item Requires sufficient FOV (may need to zoom out)
\item Assumes constant target velocity (less accurate if maneuvering)
\end{itemize}



\subsection{AMMUNITION CONSERVATION}

\textbf{Fire Discipline}:
\begin{itemize}
\item Use controlled bursts (2-5 rounds) vs. full-auto spray
\item Assess after each burst before re-engaging
\item Precision over volume
\end{itemize}

\textbf{Round Count}:
\begin{itemize}
\item Track ammunition expenditure
\item Report when low (<20\% remaining)
\item Conserve for high-priority targets
\end{itemize}



\subsection{SAFETY REMINDERS}

\textbf{ALWAYS}:
\begin{itemize}
\item $\checkmark$ Verify target before engaging (PID mandatory)
\item $\checkmark$ Check for friendly forces
\item $\checkmark$ Verify NOT in no-fire zone
\item $\checkmark$ Follow Rules of Engagement (ROE)
\item $\checkmark$ Have fire authorization (if required)
\end{itemize}

\textbf{NEVER}:
\begin{itemize}
\item $\times$ Fire without positive identification
\item $\times$ Fire into no-fire zones
\item $\times$ Fire if friendlies potentially in line of fire
\item $\times$ Fire without authorization (if required)
\item $\times$ Assume tracking is infallible (monitor track quality)
\end{itemize}
