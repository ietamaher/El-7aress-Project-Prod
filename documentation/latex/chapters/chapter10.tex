\# LESSON 10: PRACTICAL TRAINING \& EVALUATION

\textbf{Lesson Duration:} 8 hours (Hands-On Training \& Evaluation)

\textbf{Learning Objectives:}
\begin{itemize}
\item Demonstrate proficiency in all operator tasks
\item Pass written examination (80\% minimum)
\item Pass practical performance evaluation
\item Qualify for operational duty
\end{itemize}



\section{10.1 PRACTICAL TRAINING EXERCISES}

\subsection{EXERCISE 1: SYSTEM STARTUP \& HEALTH CHECK (30 min)}

\textbf{Task}: Power up system, perform pre-operation checks

\textbf{Performance Standards}:
\begin{itemize}
\item $\square$ Correct power-up sequence
\item $\square$ All devices detected and connected
\item $\square$ Pre-operation checklist 100\% complete
\item $\square$ Ready for operations within 5 minutes
\end{itemize}



\subsection{EXERCISE 2: MANUAL GIMBAL CONTROL (30 min)}

\textbf{Task}: Navigate gimbal to designated azimuth/elevation coordinates

\textbf{Performance Standards}:
\begin{itemize}
\item $\square$ Smooth, controlled movements
\item $\square$ Accuracy: ±2\degree{} azimuth, ±2\degree{} elevation
\item $\square$ Complete 5 targets within 3 minutes
\end{itemize}



\subsection{EXERCISE 3: TARGET ACQUISITION \& TRACKING (1 hour)}

\textbf{Task}: Acquire and track stationary and moving targets

\textbf{Performance Standards}:
\begin{itemize}
\item $\square$ Acquisition gate correctly sized (10-30\% margin)
\item $\square$ Lock achieved within 10 seconds
\item $\square$ Track maintained >30 seconds
\item $\square$ Clean abort on command
\end{itemize}



\subsection{EXERCISE 4: SIMULATED ENGAGEMENT (1 hour)}

\textbf{Task}: Complete engagement sequence (no live fire)

\textbf{Performance Standards}:
\begin{itemize}
\item $\square$ PID confirmed before engagement
\item $\square$ Tracking established (Active Lock)
\item $\square$ LAC activated (moving target)
\item $\square$ Safety checks complete
\item $\square$ Simulated fire (correct procedure)
\item $\square$ BDA (Battle Damage Assessment)
\end{itemize}



\subsection{EXERCISE 5: WEAPON ZEROING (1 hour)}

\textbf{Task}: Zero weapon at 200m range

\textbf{Performance Standards}:
\begin{itemize}
\item $\square$ Correct zeroing procedure
\item $\square$ Zero applied successfully
\item $\square$ Verification shot within ±5cm
\item $\square$ "Z" indicator appears on OSD
\end{itemize}



\subsection{EXERCISE 6: EMERGENCY PROCEDURES (1 hour)}

\textbf{Task}: Respond to simulated emergencies

\textbf{Scenarios}:
\begin{enumerate}
\item Wrong target identified (tracking abort)
\item Runaway gimbal (E-Stop)
\item Accidental discharge (cease fire, safe weapon)
\item Lost communication (switch backup radio)
\end{enumerate}

\textbf{Performance Standards}:
\begin{itemize}
\item $\square$ Immediate recognition of emergency
\item $\square$ Correct procedure executed
\item $\square$ Reaction time <2 seconds (critical emergencies)
\end{itemize}



\subsection{EXERCISE 7: TROUBLESHOOTING (1 hour)}

\textbf{Task}: Diagnose and resolve simulated faults

\textbf{Faults Presented}:
\begin{itemize}
\item Joystick disconnected
\item Camera offline
\item Tracking failure
\item LRF no reading
\end{itemize}

\textbf{Performance Standards}:
\begin{itemize}
\item $\square$ Correct diagnosis using SLAF method
\item $\square$ Operator-level fix applied (if applicable)
\item $\square$ Proper escalation to maintenance (if needed)
\end{itemize}



\section{10.2 WRITTEN EXAMINATION}

\textbf{Format}: 50 multiple-choice questions

\textbf{Topics Covered}:
\begin{itemize}
\item Safety procedures (10 questions)
\item Basic operation (10 questions)
\item Tracking \& engagement (10 questions)
\item Ballistics \& fire control (10 questions)
\item Emergency procedures (5 questions)
\item System status \& troubleshooting (5 questions)
\end{itemize}

\textbf{Passing Score}: 80\% (40/50 correct)

\textbf{Time Limit}: 60 minutes

\textbf{Format}: Instructor-observed practical assessment

\textbf{Evaluation Criteria}:

\textbf{Critical Errors} (automatic failure):
\begin{itemize}
\item $\times$ Safety violation (no-fire zone, friendly fire)
\item $\times$ Failed to E-Stop when required
\item $\times$ Accidental discharge (simulated)
\item $\times$ Wrong target engagement
\end{itemize}



\section{10.4 QUALIFICATION REQUIREMENTS}

\textbf{To qualify as El 7arress RCWS Operator}:

\textbf{Must pass ALL}:
\begin{itemize}
\item $\square$ Written examination (≥80\%)
\item $\square$ Practical evaluation (≥80\%)
\item $\square$ Zero critical errors
\item $\square$ Instructor recommendation
\end{itemize}

\textbf{Qualification Valid}: 12 months

\textbf{Requalification Required}:
\begin{itemize}
\item Annually (refresher training + re-test)
\item After >6 months away from system
\item After major system upgrade
\item If unsafe practices observed
\end{itemize}
